## This animation has provided a simulation of flipping coins which might
##   be helpful in understanding the concept of probability.
library(animation)
library(SNPRelate)
library(gdsfmt)
library(BiocInstaller)
library(sp)
ani.options(interval = 0.3, nmax = ifelse(interactive(), 
    50, 2))
par(mar = c(2, 3, 2, 1.5), mgp = c(1.5, 0.5, 0))
flip.coin(faces = c("Head", "Stand", "Tail"), type = "n", 
    prob = c(0.475, 0.05, 0.475), col = c(1, 2, 4))
